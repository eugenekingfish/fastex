\section*{Bracketing}

\verb|\br{contents}| \\
\textbf{Example:} \verb|$\br{x^2 + y^2}$|
$\br{x^2 + y^2}$ \\[\baselineskip]

\verb|\sbr{contents}| \\
\textbf{Example:} \verb|$\sbr{x^2 + y^2}$|
$\sbr{x^2 + y^2}$ \\[\baselineskip]

\verb|\cbr{contents}| \\
\textbf{Example:} \verb|$\cbr{x^2 + y^2}$|
$\cbr{x^2 + y^2}$ \\[\baselineskip]

\verb|\abr{contents}| \\
\textbf{Example:} \verb|$\abr{x^2 + y^2}$|
$\abr{x^2 + y^2}$ \\[\baselineskip]

\verb|\abs{contents}| \\
\textbf{Example:} \verb|$\abs{x^2 + y^2}$|
$\abs{x^2 + y^2}$ \\[\baselineskip]

\verb|\flr{contents}| \\
\textbf{Example:} \verb|$\flr{x^2 + y^2}$|
$\flr{x^2 + y^2}$ \\[\baselineskip]

\verb|\rlf{contents}| \\
\textbf{Example:} \verb|$\rlf{x^2 + y^2}$|
$\rlf{x^2 + y^2}$
